package stackOfIntegers;

public class StackOfPrimeNo {

	public static void main(String[] args) {
	
		IntStack e2=new IntStack(10);
		int cnt=0;
		int num=2;
		while(cnt<=10)
		{
			boolean flag=true;
			for(int j=2;j<num;j++)
			{
	          if(num%j==0)
	        	  flag=false;
	              break;
	        }
			if(flag==true)
			{
				e2.push(num);
				cnt++;
			}
			num++;
			
		}
		

	}
	

}
